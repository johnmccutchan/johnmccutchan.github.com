void registerSimpleAudioWithAssetManager(AudioManager audioManager, AssetManager assetManager) #

Register the simple_audio audio manager with the asset_pack asset manager. After calling this function, the asset manager will be able to load clips.

void registerSimpleAudioWithAssetManager(AudioManager audioManager,
                                        AssetManager assetManager) {
 assetManager.importers['audioclip'] = new NoopImporter();
 assetManager.loaders['audioclip'] = new _AssetLoaderClip(audioManager);
}
